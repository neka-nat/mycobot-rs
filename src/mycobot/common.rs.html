use num_traits::FromPrimitive;

#[derive(Clone)]
pub enum Angle {
    J1 = 1,
    J2 = 2,
    J3 = 3,
    J4 = 4,
    J5 = 5,
    J6 = 6,
}

#[derive(Clone)]
pub enum Coord {
    X = 1,
    Y = 2,
    Z = 3,
    Rx = 4,
    Ry = 5,
    Rz = 6,
}

impl FromPrimitive for Coord {
    fn from_i64(n: i64) -> Option<Coord> {
        match n {
            1 => Some(Coord::X),
            2 => Some(Coord::Y),
            3 => Some(Coord::Z),
            4 => Some(Coord::Rx),
            5 => Some(Coord::Ry),
            6 => Some(Coord::Rz),
            _ => None,
        }
    }
    fn from_u64(n: u64) -> Option<Coord> {
        match n {
            1 => Some(Coord::X),
            2 => Some(Coord::Y),
            3 => Some(Coord::Z),
            4 => Some(Coord::Rx),
            5 => Some(Coord::Ry),
            6 => Some(Coord::Rz),
            _ => None,
        }
    }
}
